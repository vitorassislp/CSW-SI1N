import numpy as np
import pandas as pd
from scipy import stats

# Criando um DataFrame de exemplo
data = {'valor': [10, 12, 13, 15, 100, 17, 18, 20, 19, 200]}
df = pd.DataFrame(data)

# Calculando o Z-score
z_scores = np.abs(stats.zscore(df['valor']))

# Definindo um limite para o Z-score (por exemplo, 3)
threshold = 3

# Removendo os outliers
df_clean = df[z_scores < threshold]

print(df_clean)
